Étoile centrale dorée

Dessin d'une étoile à 5 branches avec la fonction drawFivePointedStar().

function drawFivePointedStar(ctx, cx, cy, outerRadius, innerRadius) {
  ctx.beginPath();
  var angle = Math.PI / 2;
  var angleStep = Math.PI / 5;
  for (var i = 0; i < 10; i++) {
    var radius = (i % 2 === 0) ? outerRadius : innerRadius;
    var x = cx + Math.cos(angle) * radius;
    var y = cy - Math.sin(angle) * radius;
    if (i === 0) ctx.moveTo(x, y);
    else ctx.lineTo(x, y);
    angle += angleStep;
  }
  ctx.closePath();
}

drawFivePointedStar(ctx, 200, 150, 30, 15);
ctx.fillStyle = '#FFD700';
ctx.fill();
