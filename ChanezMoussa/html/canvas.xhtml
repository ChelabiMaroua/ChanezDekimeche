Explication de la Construction

Étapes détaillées de création du logo

1
Initialisation du Canvas

Création du contexte 2D et définition des dimensions.

var canvas = document.getElementById('logoCanvas'); var ctx = canvas.getContext('2d'); canvas.width = 400; canvas.height = 300;
2
Fond avec dégradé

Création d'un fond vert avec dégradé linéaire.

var gradient = ctx.createLinearGradient(0, 0, 400, 300); gradient.addColorStop(0, '#006233'); gradient.addColorStop(1, '#008044'); ctx.fillStyle = gradient; ctx.fillRect(0, 0, 400, 300);
3
Cercles concentriques

Dessin de trois cercles pour l'effet "cible".

// Cercle extérieur blanc ctx.beginPath(); ctx.arc(200, 150, 80, 0, Math.PI * 2); ctx.strokeStyle = '#FFFFFF'; ctx.lineWidth = 3; ctx.stroke(); // Cercle moyen rouge ctx.beginPath(); ctx.arc(200, 150, 60, 0, Math.PI * 2); ctx.strokeStyle = '#DA251D'; ctx.lineWidth = 3; ctx.stroke(); // Cercle intérieur blanc ctx.beginPath(); ctx.arc(200, 150, 40, 0, Math.PI * 2); ctx.strokeStyle = '#FFFFFF'; ctx.lineWidth = 3; ctx.stroke();
4
Étoile centrale dorée

Dessin d'une étoile à 5 branches avec la fonction drawFivePointedStar().

function drawFivePointedStar(ctx, cx, cy, outerRadius, innerRadius) { ctx.beginPath(); var angle = Math.PI / 2; var angleStep = Math.PI / 5; for (var i = 0; i < 10; i++) { var radius = (i % 2 === 0) ? outerRadius : innerRadius; var x = cx + Math.cos(angle) * radius; var y = cy - Math.sin(angle) * radius; if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y); angle += angleStep; } ctx.closePath(); } drawFivePointedStar(ctx, 200, 150, 30, 15); ctx.fillStyle = '#FFD700'; ctx.fill();
